def _force_randomize_questions(self, questions):
    """–ü–µ—Ä–µ–º–µ—à–∏–≤–∞–µ—Ç –≤–∞—Ä–∏–∞–Ω—Ç—ã –æ—Ç–≤–µ—Ç–æ–≤ (API –Ω–µ –≤—Å–µ–≥–¥–∞ —Å–ª–µ–¥—É–µ—Ç –∏–Ω—Å—Ç—Ä—É–∫—Ü–∏—è–º)"""

        # –ü–æ–ª—É—á–∞–µ–º –ø—Ä–∞–≤–∏–ª—å–Ω—ã–π –æ—Ç–≤–µ—Ç

        randomized_questions.append(question)
    
    return randomized_questions
